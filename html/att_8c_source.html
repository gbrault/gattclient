282 static bool encode_pdu(struct bt_att *att, struct att_send_op *op,
283  const void *pdu, uint16_t length)
284 {
285  uint16_t pdu_len = 1;
286  struct sign_info *sign = att->local_sign;
287  uint32_t sign_cnt;
288 
289  if (sign && (op->opcode & ATT_OP_SIGNED_MASK))
290  pdu_len += BT_ATT_SIGNATURE_LEN;
291 
292  if (length && pdu)
293  pdu_len += length;
294 
295  if (pdu_len > att->mtu)
296  return false;
297 
298  op->len = pdu_len;
299  op->pdu = malloc(op->len);
300  if (!op->pdu)
301  return false;
302 
303  ((uint8_t *) op->pdu)[0] = op->opcode;
304  if (pdu_len > 1)
305  memcpy(op->pdu + 1, pdu, length);
306 
307  if (!sign || !(op->opcode & ATT_OP_SIGNED_MASK))
308  return true;
309 
310  if (!sign->counter(&sign_cnt, sign->user_data))
311  goto fail;
312 
313  if ((bt_crypto_sign_att(att->crypto, sign->key, op->pdu, 1 + length,
314  sign_cnt, &((uint8_t *) op->pdu)[1 + length])))
315  return true;
316 
317  util_debug(att->debug_callback, att->debug_data,
318  "ATT unable to generate signature");
319 
320 fail:
321  free(op->pdu);
322  return false;
323 }

325 static struct att_send_op *create_att_send_op(struct bt_att *att,
326  uint8_t opcode,
327  const void *pdu,
328  uint16_t length,
329  bt_att_response_func_t callback,
330  void *user_data,
331  bt_att_destroy_func_t destroy)
332 {
333  struct att_send_op *op;
334  enum att_op_type op_type;
335 
336  if (length && !pdu)
337  return NULL;
338 
339  op_type = get_op_type(opcode);
340  if (op_type == ATT_OP_TYPE_UNKNOWN)
341  return NULL;
342 
343  /* If the opcode corresponds to an operation type that does not elicit a
344  * response from the remote end, then no callback should have been
345  * provided, since it will never be called.
346  */
347  if (callback && op_type != ATT_OP_TYPE_REQ && op_type != ATT_OP_TYPE_IND)
348  return NULL;
349 
350  /* Similarly, if the operation does elicit a response then a callback
351  * must be provided.
352  */
353  if (!callback && (op_type == ATT_OP_TYPE_REQ || op_type == ATT_OP_TYPE_IND))
354  return NULL;
355 
356  op = new0(struct att_send_op, 1);
357  if (!op)
358  return NULL;
359 
360  op->type = op_type;
361  op->opcode = opcode;
362  op->callback = callback;
363  op->destroy = destroy;
364  op->user_data = user_data;
365 
366  if (!encode_pdu(att, op, pdu, length)) {
367  free(op);
368  return NULL;
369  }
370 
371  return op;
372 }

374 static struct att_send_op *pick_next_send_op(struct bt_att *att)
375 {
376  struct att_send_op *op;
377 
378  /* See if any operations are already in the write queue */
379  op = queue_pop_head(att->write_queue);
380  if (op)
381  return op;
382 
383  /* If there is no pending request, pick an operation from the
384  * request queue.
385  */
386  if (!att->pending_req) {
387  op = queue_pop_head(att->req_queue);
388  if (op)
389  return op;
390  }
391 
392  /* There is either a request pending or no requests queued. If there is
393  * no pending indication, pick an operation from the indication queue.
394  */
395  if (!att->pending_ind) {
396  op = queue_pop_head(att->ind_queue);
397  if (op)
398  return op;
399  }
400 
401  return NULL;
402 }

421  }
422 
423  if (!op)

432  op->timeout_id = 0;

437  * This should trigger an io disconnect event which will clean up the
438  * io and notify the upper layer.
439  */

442  return false;

452 static bool can_write_data(struct io *io, void *user_data)
453 {
454  struct bt_att *att = user_data;
455  struct att_send_op *op;
456  struct timeout_data *timeout;
457  ssize_t ret;
458  struct iovec iov;
459 
460  op = pick_next_send_op(att);
461  if (!op)
462  return false;
463 
464  iov.iov_base = op->pdu;
465  iov.iov_len = op->len;
466 
467  ret = io_send(io, &iov, 1);
468  if (ret < 0) {
469  util_debug(att->debug_callback, att->debug_data,
470  "write failed: %s", strerror(-ret));
471  if (op->callback)
472  op->callback(BT_ATT_OP_ERROR_RSP, NULL, 0,
473  op->user_data);
474 
475  destroy_att_send_op(op);
476  return true;
477  }
478 
479  util_debug(att->debug_callback, att->debug_data,
480  "ATT op 0x%02x", op->opcode);
481 
482  util_hexdump('<', op->pdu, ret, att->debug_callback, att->debug_data);
483 
484  /* Based on the operation type, set either the pending request or the
485  * pending indication. If it came from the write queue, then there is
486  * no need to keep it around.
487  */
488  switch (op->type) {
489  case ATT_OP_TYPE_REQ:
490  att->pending_req = op;
491  break;
492  case ATT_OP_TYPE_IND:
493  att->pending_ind = op;
494  break;
495  case ATT_OP_TYPE_RSP:
496  /* Set in_req to false to indicate that no request is pending */
497  att->in_req = false;
498  /* Fall through to the next case */
499  case ATT_OP_TYPE_CMD:
500  case ATT_OP_TYPE_NOT:
501  case ATT_OP_TYPE_CONF:
502  case ATT_OP_TYPE_UNKNOWN:
503  default:
504  destroy_att_send_op(op);
505  return true;
506  }
507 
508  timeout = new0(struct timeout_data, 1);
509  if (!timeout)
510  return true;
511 
512  timeout->att = att;
513  timeout->id = op->id;
514  op->timeout_id = timeout_add(ATT_TIMEOUT_INTERVAL, timeout_cb,
515  timeout, free);
516 
517  /* Return true as there may be more operations ready to write. */
518  return true;
519 }

636 static void handle_rsp(struct bt_att *att, uint8_t opcode, uint8_t *pdu,
637  ssize_t pdu_len)
638 {
639  struct att_send_op *op = att->pending_req;
640  uint8_t req_opcode;
641  uint8_t rsp_opcode;
642  uint8_t *rsp_pdu = NULL;
643  uint16_t rsp_pdu_len = 0;
644 
645  /*
646  * If no request is pending, then the response is unexpected. Disconnect
647  * the bearer.
648  */
649  if (!op) {
650  util_debug(att->debug_callback, att->debug_data,
651  "Received unexpected ATT response");
652  io_shutdown(att->io);
653  return;
654  }
655 
656  /*
657  * If the received response doesn't match the pending request, or if
658  * the request is malformed, end the current request with failure.
659  */
660  if (opcode == BT_ATT_OP_ERROR_RSP) {
661  /* Return if error response cause a retry */
662  if (handle_error_rsp(att, pdu, pdu_len, &req_opcode)) {
663  wakeup_writer(att);
664  return;
665  }
666  } else if (!(req_opcode = get_req_opcode(opcode)))
667  goto fail;
668 
669  if (req_opcode != op->opcode)
670  goto fail;
671 
672  rsp_opcode = opcode;
673 
674  if (pdu_len > 0) {
675  rsp_pdu = pdu;
676  rsp_pdu_len = pdu_len;
677  }
678 
679  goto done;
680 
681 fail:
682  util_debug(att->debug_callback, att->debug_data,
683  "Failed to handle response PDU; opcode: 0x%02x", opcode);
684 
685  rsp_opcode = BT_ATT_OP_ERROR_RSP;
686 
687 done:
688  if (op->callback)
689  op->callback(rsp_opcode, rsp_pdu, rsp_pdu_len, op->user_data);
690 
691  destroy_att_send_op(op);
692  att->pending_req = NULL;
693 
694  wakeup_writer(att);
695 }

788 static void handle_notify(struct bt_att *att, uint8_t opcode, uint8_t *pdu,
789  ssize_t pdu_len)
790 {
791  const struct queue_entry *entry;
792  bool found;
793 
794  if ((opcode & ATT_OP_SIGNED_MASK) && !att->ext_signed) {
795  if (!handle_signed(att, opcode, pdu, pdu_len))
796  return;
797  pdu_len -= BT_ATT_SIGNATURE_LEN;
798  }
799 
800  bt_att_ref(att);
801 
802  found = false;
803  entry = queue_get_entries(att->notify_list);
804 
805  while (entry) {
806  struct att_notify *notify = entry->data;
807 
808  entry = entry->next;
809 
810  if (!opcode_match(notify->opcode, opcode))
811  continue;
812 
813  found = true;
814 
815  if (notify->callback)
816  notify->callback(opcode, pdu, pdu_len,
817  notify->user_data);
818 
819  /* callback could remove all entries from notify list */
820  if (queue_isempty(att->notify_list))
821  break;
822  }
823 
824  /*
825  * If this was a request and no handler was registered for it, respond
826  * with "Not Supported"
827  */
828  if (!found && get_op_type(opcode) == ATT_OP_TYPE_REQ)
829  respond_not_supported(att, opcode);
830 
831  bt_att_unref(att);
832 }

834 static bool can_read_data(struct io *io, void *user_data)
835 {
836  struct bt_att *att = user_data;
837  uint8_t opcode;
838  uint8_t *pdu;
839  ssize_t bytes_read;
840 
841  bytes_read = read(att->fd, att->buf, att->mtu);
842  if (bytes_read < 0)
843  return false;
844 
845  util_hexdump('>', att->buf, bytes_read,
846  att->debug_callback, att->debug_data);
847 
848  if (bytes_read < ATT_MIN_PDU_LEN)
849  return true;
850 
851  pdu = att->buf;
852  opcode = pdu[0];
853 
854  bt_att_ref(att);
855 
856  /* Act on the received PDU based on the opcode type */
857  switch (get_op_type(opcode)) {
858  case ATT_OP_TYPE_RSP:
859  util_debug(att->debug_callback, att->debug_data,
860  "ATT response received: 0x%02x", opcode);
861  handle_rsp(att, opcode, pdu + 1, bytes_read - 1);
862  break;
863  case ATT_OP_TYPE_CONF:
864  util_debug(att->debug_callback, att->debug_data,
865  "ATT confirmation received: 0x%02x", opcode);
866  handle_conf(att, pdu + 1, bytes_read - 1);
867  break;
868  case ATT_OP_TYPE_REQ:
869  /*
870  * If a request is currently pending, then the sequential
871  * protocol was violated. Disconnect the bearer, which will
872  * promptly notify the upper layer via disconnect handlers.
873  */
874  if (att->in_req) {
875  util_debug(att->debug_callback, att->debug_data,
876  "Received request while another is "
877  "pending: 0x%02x", opcode);
878  io_shutdown(att->io);
879  bt_att_unref(att);
880 
881  return false;
882  }
883 
884  att->in_req = true;
885 
886  /* Fall through to the next case */
887  case ATT_OP_TYPE_CMD:
888  case ATT_OP_TYPE_NOT:
889  case ATT_OP_TYPE_UNKNOWN:
890  case ATT_OP_TYPE_IND:
891  default:
892  /* For all other opcodes notify the upper layer of the PDU and
893  * let them act on it.
894  */
895  util_debug(att->debug_callback, att->debug_data,
896  "ATT PDU received: 0x%02x", opcode);
897  handle_notify(att, opcode, pdu + 1, bytes_read - 1);
898  break;
899  }
900 
901  bt_att_unref(att);
902 
903  return true;
904 }
