277 bool bt_crypto_sign_att(struct bt_crypto *crypto, const uint8_t key[16],
278  const uint8_t *m, uint16_t m_len,
279  uint32_t sign_cnt, uint8_t signature[12])
280 {
281  int fd;
282  int len;
283  uint8_t tmp[16], out[16];
284  uint16_t msg_len = m_len + sizeof(uint32_t);
285  uint8_t msg[msg_len];
286  uint8_t msg_s[msg_len];
287 
288  if (!crypto)
289  return false;
290 
291  memset(msg, 0, msg_len);
292  memcpy(msg, m, m_len);
293 
294  /* Add sign_counter to the message */
295  put_le32(sign_cnt, msg + m_len);
296 
297  /* The most significant octet of key corresponds to key[0] */
298  swap_buf(key, tmp, 16);
299 
300  fd = alg_new(crypto->cmac_aes, tmp, 16);
301  if (fd < 0)
302  return false;
303 
304  /* Swap msg before signing */
305  swap_buf(msg, msg_s, msg_len);
306 
307  len = send(fd, msg_s, msg_len, 0);
308  if (len < 0) {
309  close(fd);
310  return false;
311  }
312 
313  len = read(fd, out, 16);
314  if (len < 0) {
315  close(fd);
316  return false;
317  }
318 
319  close(fd);
320 
321  /*
322  * As to BT spec. 4.1 Vol[3], Part C, chapter 10.4.1 sign counter should
323  * be placed in the signature
324  */
325  put_be32(sign_cnt, out + 8);
326 
327  /*
328  * The most significant octet of hash corresponds to out[0] - swap it.
329  * Then truncate in most significant bit first order to a length of
330  * 12 octets
331  */
332  swap_buf(out, tmp, 16);
333  memcpy(signature, tmp + 4, 12);
334 
335  return true;
336 }
337 /*
338  * Security function e
339  *
340  * Security function e generates 128-bit encryptedData from a 128-bit key
341  * and 128-bit plaintextData using the AES-128-bit block cypher:
342  *
343  * encryptedData = e(key, plaintextData)
344  *
345  * The most significant octet of key corresponds to key[0], the most
346  * significant octet of plaintextData corresponds to in[0] and the
347  * most significant octet of encryptedData corresponds to out[0].
348  *
349  */
350 bool bt_crypto_e(struct bt_crypto *crypto, const uint8_t key[16],
351  const uint8_t plaintext[16], uint8_t encrypted[16])
352 {
353  uint8_t tmp[16], in[16], out[16];
354  int fd;
355 
356  if (!crypto)
357  return false;
358 
359  /* The most significant octet of key corresponds to key[0] */
360  swap_buf(key, tmp, 16);
361 
362  fd = alg_new(crypto->ecb_aes, tmp, 16);
363  if (fd < 0)
364  return false;
365 
366 
367  /* Most significant octet of plaintextData corresponds to in[0] */
368  swap_buf(plaintext, in, 16);
369 
370  if (!alg_encrypt(fd, in, 16, out, 16)) {
371  close(fd);
372  return false;
373  }
374 
375  /* Most significant octet of encryptedData corresponds to out[0] */
376  swap_buf(out, encrypted, 16);
377 
378  close(fd);
379 
380  return true;
381 }

383 /*
384  * Random Address Hash function ah
385  *
386  * The random address hash function ah is used to generate a hash value
387  * that is used in resolvable private addresses.
388  *
389  * The following are inputs to the random address hash function ah:
390  *
391  * k is 128 bits
392  * r is 24 bits
393  * padding is 104 bits
394  *
395  * r is concatenated with padding to generate r' which is used as the
396  * 128-bit input parameter plaintextData to security function e:
397  *
398  * r' = padding || r
399  *
400  * The least significant octet of r becomes the least significant octet
401  * of r and the most significant octet of padding becomes the most
402  * significant octet of r'.
403  *
404  * For example, if the 24-bit value r is 0x423456 then r' is
405  * 0x00000000000000000000000000423456.
406  *
407  * The output of the random address function ah is:
408  *
409  * ah(k, r) = e(k, r') mod 2^24
410  *
411  * The output of the security function e is then truncated to 24 bits by
412  * taking the least significant 24 bits of the output of e as the result
413  * of ah.
414  */
415 bool bt_crypto_ah(struct bt_crypto *crypto, const uint8_t k[16],
416  const uint8_t r[3], uint8_t hash[3])
417 {
418  uint8_t rp[16];
419  uint8_t encrypted[16];
420 
421  if (!crypto)
422  return false;
423 
424  /* r' = padding || r */
425  memcpy(rp, r, 3);
426  memset(rp + 3, 0, 13);
427 
428  /* e(k, r') */
429  if (!bt_crypto_e(crypto, k, rp, encrypted))
430  return false;
431 
432  /* ah(k, r) = e(k, r') mod 2^24 */
433  memcpy(hash, encrypted, 3);
434 
435  return true;
436 }

456 /*
457  * Confirm value generation function c1
458  *
459  * During the pairing process confirm values are exchanged. This confirm
460  * value generation function c1 is used to generate the confirm values.
461  *
462  * The following are inputs to the confirm value generation function c1:
463  *
464  * k is 128 bits
465  * r is 128 bits
466  * pres is 56 bits
467  * preq is 56 bits
468  * iat is 1 bit
469  * ia is 48 bits
470  * rat is 1 bit
471  * ra is 48 bits
472  * padding is 32 bits of 0
473  *
474  * iat is concatenated with 7-bits of 0 to create iat' which is 8 bits
475  * in length. iat is the least significant bit of iat'
476  *
477  * rat is concatenated with 7-bits of 0 to create rat' which is 8 bits
478  * in length. rat is the least significant bit of rat'
479  *
480  * pres, preq, rat' and iat' are concatenated to generate p1 which is
481  * XORed with r and used as 128-bit input parameter plaintextData to
482  * security function e:
483  *
484  * p1 = pres || preq || rat' || iat'
485  *
486  * The octet of iat' becomes the least significant octet of p1 and the
487  * most significant octet of pres becomes the most significant octet of
488  * p1.
489  *
490  * ra is concatenated with ia and padding to generate p2 which is XORed
491  * with the result of the security function e using p1 as the input
492  * paremter plaintextData and is then used as the 128-bit input
493  * parameter plaintextData to security function e:
494  *
495  * p2 = padding || ia || ra
496  *
497  * The least significant octet of ra becomes the least significant octet
498  * of p2 and the most significant octet of padding becomes the most
499  * significant octet of p2.
500  *
501  * The output of the confirm value generation function c1 is:
502  *
503  * c1(k, r, preq, pres, iat, rat, ia, ra) = e(k, e(k, r XOR p1) XOR p2)
504  *
505  * The 128-bit output of the security function e is used as the result
506  * of confirm value generation function c1.
507  */
508 bool bt_crypto_c1(struct bt_crypto *crypto, const uint8_t k[16],
509  const uint8_t r[16], const uint8_t pres[7],
510  const uint8_t preq[7], uint8_t iat,
511  const uint8_t ia[6], uint8_t rat,
512  const uint8_t ra[6], uint8_t res[16])
513 {
514  uint8_t p1[16], p2[16];
515 
516  /* p1 = pres || preq || _rat || _iat */
517  p1[0] = iat;
518  p1[1] = rat;
519  memcpy(p1 + 2, preq, 7);
520  memcpy(p1 + 9, pres, 7);
521 
522  /* p2 = padding || ia || ra */
523  memcpy(p2, ra, 6);
524  memcpy(p2 + 6, ia, 6);
525  memset(p2 + 12, 0, 4);
526 
527  /* res = r XOR p1 */
528  u128_xor(r, p1, res);
529 
530  /* res = e(k, res) */
531  if (!bt_crypto_e(crypto, k, res, res))
532  return false;
533 
534  /* res = res XOR p2 */
535  u128_xor(res, p2, res);
536 
537  /* res = e(k, res) */
538  return bt_crypto_e(crypto, k, res, res);
539 }
